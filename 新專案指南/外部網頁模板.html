External App

外部 App

這是一份外部 App 起手模板。請先複製到新專案，再建立 app-card.json、version.js 和 assets/js/app.js。

New Project Checklist

開新外部專案還需要做的事

  1. 建立新的外部專案資料夾，例如 your-app。
  2. 依本頁 index.html head / body script 範本建立外部專案的 index.html，不要整檔複製此 viewer 文件。
  3. 在 HTML 裡只設定 runtime resolver、SKHPS_ENTRY_BASE_URL、SKHPS_ENTRY_VERSION、SKHPS_APP_CARD_URL。
  4. 在 app-card.json 裡設定 appId、title、group、order、href、afterScripts、registerExternalApp。
  5. 確認 data-skhps-loading-tasks 內的 app-ready 和 app.js 呼叫的 loading task 名稱一致。
  6. 建立 assets/js/app.js，內容可先使用下方最小範本。
  7. 建立 version.js，透過 window.SKHPS_VERSION 提供目前外部 App 版本。
  8. 外部 App 只直接載入 app-entry.js；不要直接載入 external-app-loader.js。
  9. 不要在外部 App 自己再載入 config.js、loading-gate.js、backend-client.js、css-sheet-runtime.js、header.js、footer.js。
  10. 如果需要後端功能，新增 Apps Script action，前端用 SKHPSBackend.call("actionName", payload)。
  11. 部署後確認 app-entry.js、external-app-loader.js、header、footer、CSS runtime、backend、afterScripts 都有載入成功。
  12. CSS 如果還在吃 cache，可點 footer 的 CSS 狀態燈強迫重讀 Sheet。

index.html head

head 最小呼叫範本

<script>
  (function () {
    "use strict";

    var host = String(location.hostname || "").toLowerCase();
    var params = new URLSearchParams(location.search || "");
    var runtime = String(params.get("skhpsRuntime") || "").trim();
    var isLocal = host === "127.0.0.1" || host === "localhost" || host === "";
    var allowedRuntime = { "local-dev": true, dev: true, prod: true };

    function runtimeBaseUrl(value) {
      if (value === "local-dev") return "http://127.0.0.1:5500/skhpsv2/";
      if (value === "dev") return "https://dev-skhps.jonaminz.com/";
      return "https://skhps.jonaminz.com/";
    }

    function currentRuntime() {
      if (isLocal) return "local-dev";
      if (allowedRuntime[runtime]) return runtime;
      return "prod";
    }

    var resolvedRuntime = currentRuntime();

    window.SKHPS_ENTRY_VERSION = "202606131435";
    window.SKHPS_ENTRY_BASE_URL = runtimeBaseUrl(resolvedRuntime);
    window.SKHPS_APP_CARD_URL = "app-card.json";

    document.documentElement.setAttribute("data-skhps-runtime", resolvedRuntime);

    document.write(
      '<link rel="stylesheet" href="' +
      window.SKHPS_ENTRY_BASE_URL +
      'assets/css/skhps-loading.css?v=' +
      encodeURIComponent(window.SKHPS_ENTRY_VERSION) +
      '">'
    );
  })();
</script>

assets/js/app.js

最小 app.js 範本

(function () {
  "use strict";

  var APP_READY_TASK = "app-ready";

  function loadingDone(task) {
    if (window.SKHPSLoading && typeof window.SKHPSLoading.done === "function") {
      window.SKHPSLoading.done(task);
      return;
    }

    document.documentElement.classList.remove("skhps-css-loading");
    document.documentElement.classList.remove("skhps-loading");
  }

  function loadingFail(task, error) {
    if (window.SKHPSLoading && typeof window.SKHPSLoading.fail === "function") {
      window.SKHPSLoading.fail(task, error);
      return;
    }

    document.documentElement.classList.remove("skhps-css-loading");
    document.documentElement.classList.remove("skhps-loading");
  }

  function init() {
    try {
      /*
        在這裡初始化你的外部 App。
        如果需要後端資料，可以用：
        SKHPSBackend.call("actionName", {});
      */
      loadingDone(APP_READY_TASK);
    } catch (error) {
      console.error(error);
      loadingFail(APP_READY_TASK, error);
    }
  }

  if (document.readyState === "loading") {
    document.addEventListener("DOMContentLoaded", init);
  } else {
    init();
  }
})();

app-card.json

app-card.json 範本

app-entry.js 會讀取 app-card.json，建立 SKHPS_APP_CONFIG / SKHPS_APP_ENV，再交給 external-app-loader.js 載入共用 runtime。 

{
  "appId": "your-app-id",
  "title": "外部 App",
  "group": "未分類",
  "order": 9999,
  "registerExternalApp": true,
  "versionUrl": "version.js",
  "afterScripts": [
    "assets/js/app.js"
  ],
  "href": {
    "local-dev": "http://127.0.0.1:5500/your-app/",
    "dev": "https://your-app.example.com/",
    "prod": "https://your-app.example.com/"
  }
}

index.html body script

body 最小載入 app-entry.js

<script>
  (function () {
    "use strict";

    var script = document.createElement("script");
    script.src =
      window.SKHPS_ENTRY_BASE_URL +
      "assets/js/app-entry.js?v=" +
      encodeURIComponent(window.SKHPS_ENTRY_VERSION);
    script.async = false;
    document.head.appendChild(script);
  })();
</script>
